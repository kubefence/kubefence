```toml
# CRI-O NRI configuration for nono-nri plugin
# Install as: /etc/crio/crio.conf.d/10-nono-nri.conf
#
# For pre-installed plugin mode (DEPL-03), also copy the plugin binary:
#   cp 10-nono-nri /opt/nri/plugins/10-nono-nri
#   chmod +x /opt/nri/plugins/10-nono-nri
#
# Requires CRI-O >= 1.27

enable_nri = true
nri_plugin_dir = "/opt/nri/plugins"
nri_plugin_config_dir = "/etc/nri/conf.d"
nri_listen = "/var/run/nri/nri.sock"
nri_disable_connections = false
nri_plugin_registration_timeout = "5s"
nri_plugin_request_timeout = "2s"
```

**deploy/containerd-config.toml** — Containerd NRI config snippet:
```toml
# Containerd NRI configuration for nono-nri plugin
# Merge into: /etc/containerd/config.toml
#
# IMPORTANT: The key MUST include the trailing ".nri" suffix.
# Using [plugins."io.containerd.nri.v1"] (without .nri) silently fails.
#
# containerd 2.x has NRI enabled by default; this block customizes paths.
# containerd 1.7.x requires this block to enable NRI.

[plugins."io.containerd.nri.v1.nri"]
  disable = false
  disable_connections = false
  plugin_config_path = "/etc/nri/conf.d"
  plugin_path = "/opt/nri/plugins"
  plugin_registration_timeout = "5s"
  plugin_request_timeout = "2s"
  socket_path = "/var/run/nri/nri.sock"
```

**deploy/10-nono-nri.toml.example** — Plugin TOML config example:
```toml
# nono-nri plugin configuration
# Install as: /etc/nri/conf.d/10-nono-nri.toml
#
# runtime_classes: list of RuntimeClass metadata.name values to intercept.
#   Pods with runtimeClassName matching any entry get nono injection.
#   All other pods are skipped (no-op).
#
# default_profile: nono profile used when pod lacks nono.sh/profile annotation.
#
# nono_bin_path: absolute path to the nono binary on the HOST node.
#   The DaemonSet init container copies nono here from the image.
#   This path is bind-mounted into sandboxed containers.
#
# socket_path: NRI socket path. Empty string uses the NRI default
#   (/var/run/nri/nri.sock).

runtime_classes = ["nono-sandbox"]
default_profile = "default"
nono_bin_path = "/opt/nono-nri/nono"
socket_path = ""
```
